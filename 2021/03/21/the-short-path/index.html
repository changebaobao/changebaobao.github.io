Dijkstra

#include <bits/stdc++.h>
using namespace std;
int main()
{
    int e[10][10], dis[10], book[10], n, m, t1, t2, t3, u, minn;
    int inf = 999999;
    //初始化
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        for (int j = 0; j <= n; j++) {
            if (i == j) {
                e[i][j] = 0;
            } else {
                e[i][j] = inf;
            }
        }
    }

    for (int i = 1; i <= m; i++) {
        cin >> t1 >> t2 >> t3;
        e[t1][t2] = t3;
    }

    for (int i = 1; i <= n; i++) {
        dis[i] = e[1][i];
    }

    for (int i = 1; i <= n; i++) {
        book[i] = 0;
    }
    //核心语句
    book[1] = 1;
    for (int i = 1; i <= n - 1; i++) {
        minn = inf;
        for (int j = 0; j <= n; j++) {
            if (book[j] == 0 && dis[j] < minn) {
                minn = dis[j];
                u = j;
            }
        }
        book[u] = 1;
        for (int v = 1; v <= n; v++) {
            if (e[u][v] < inf) {
                dis[v] = min(dis[v], dis[u] + e[u][v]);
            }
        }
    }
    for (int i = 1; i <= n; i++) {
        cout << dis[i] << " ";
    }
    return 0;
}
