#include <bits/stdc++.h>
using namespace std;
int p[1010];
void init() { 
for (int i = 0; i < 1010; i++) {
        p[i] = i;
    }
}
int find(int x)
{
    return x == p[x] ? x : find(p[x]);
}
void Union(int a, int b)
{
    int t1 = find(a);
    int t2 = find(b);
    if (t1!=t2) {
        p[t1] = t2;
    }
}
int main()
{
    int t;
    cin >> t;
    int n, m, x, y, ans;
    while (t--) {
        cin >> n >> m;
        init();
        for (int i = 0; i < m; i++) {
            cin >> x >> y;
            Union(x, y);
        }
        ans = 0;
        for (int i = 1; i <= n; i++) {
            if (p[i]==i) {
                ans++;
            }
        }
        cout << ans << endl;
    }

    return 0;
}
