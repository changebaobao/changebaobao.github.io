模板题http://acm.hdu.edu.cn/showproblem.php?pid=1130

#include <bits/stdc++.h>
using namespace std;
const int LEN = 1005;
int num[LEN][LEN];
int c[LEN][LEN];
void init()
{
    memset(num, 0, sizeof(num));
    num[1][1] = 1;
    for (int i = 2; i <= 1001; i++) {
        for (int j = 1; j <= 1001; j++) {
            num[i][j] += num[i - 1][j] * (4 * i - 2);
        }
        for (int j = 1; j <= 1001; j++) {
            num[i][j + 1] += num[i][j] / 10;
            num[i][j] = num[i][j] % 10;
        }
        for (int j = 1001; j > 0; j--) {
            num[i][j - 1] += num[i][j] % (i + 1) * 10;
            num[i][j] /= i + 1;
        }
    }
}
int main()
{
    init();
    int n;
    while (~scanf_s("%d", &n)) {
        int end = 1000;
        while (num[n][end] == 0) {
            end--;
        }

        while (end > 0) {
            putchar('0' + num[n][end--]);
        }
        cout << endl;
    }
    return 0;
}
